function humanReadable(inputSec) {
  let hours, min, sec
  hours = Math.floor(inputSec/3600) + ''
  hours = !hours[1] ? `0${hours}` : hours
  min = Math.floor((inputSec-hours*3600)/60) + ''
  min = !min[1] ? `0${min}` : min
  sec = inputSec-hours*3600-min*60 + ''
  sec = !sec[1] ? `0${sec}` : sec
  return `${hours}:${min}:${sec}`
}
